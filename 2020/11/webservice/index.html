CXF入门

CXF是apache旗下的开源框架，由Celtix + XFire这两门经典的框架合成，是一套非常流行的web service框架。

它提供了JAX-WS的全面支持，并且可以根据实际项目的需要，采用代码优先（Code First）或者 WSDL 优先（WSDL First）来轻松地实现 Web Services 的发布和使用，同时它能与spring进行完美结合。

CommonWebService
package com.example.serurity.service;

import org.springframework.stereotype.Component;

import javax.jws.WebMethod;
import javax.jws.WebParam;
import javax.jws.WebResult;
import javax.jws.WebService;

/**
 * @description： test
 * @author: WangXiaoWei xiaowei.wang01@hand-china.com
 * @date: 2019/7/2
 **/
@WebService(name = "CommonWebService",
targetNamespace = "http://serurity.example.com/")
@Component
public interface CommonWebService {
    @WebMethod
    @WebResult(name = "String",targetNamespace = "")
    String sayHello(@WebParam(name = "userName") String userName);
}
CommonWebServiceImpl
package com.example.serurity.service;

import org.springframework.stereotype.Component;

import javax.jws.WebService;

/**
 * @description：
 * @author: WangXiaoWei xiaowei.wang01@hand-china.com
 * @date: 2019/7/2
 **/
@WebService(name = "CommonWebService",
        targetNamespace = "http://serurity.example.com/",
        endpointInterface = "com.example.serurity.service.CommonWebService"
)
@Component
public class CommonWebServiceImpl implements CommonWebService {
    @Override
    public String sayHello(String userName) {
        return "Hello"+userName;
    }
}
CxfConfig
package com.example.serurity.config;

import com.example.serurity.service.CommonWebService;
import org.apache.cxf.Bus;
import org.apache.cxf.jaxws.EndpointImpl;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;

import javax.xml.ws.Endpoint;

/**
 * @description： cxf config
 * @author: WangXiaoWei xiaowei.wang01@hand-china.com
 * @date: 2019/7/2
 **/
@Configuration
public class CxfConfig {
    @Autowired
    private Bus bus;

    @Autowired
    CommonWebService commonService;

    /** JAX-WS **/
    @Bean
    public Endpoint endpoint() {
        EndpointImpl endpoint = new EndpointImpl(bus, commonService);
        endpoint.publish("/CommonService");
        return endpoint;
    }
}

创建WebServiceClient 调用

package com.example.serurity.client;

import com.example.serurity.service.CommonWebService;
import org.apache.cxf.endpoint.Client;
import org.apache.cxf.jaxws.JaxWsProxyFactoryBean;
import org.apache.cxf.jaxws.endpoint.dynamic.JaxWsDynamicClientFactory;

/**
 * @description： test
 * @author: WangXiaoWei xiaowei.wang01@hand-china.com
 * @date: 2019/7/2
 **/
public class WebServiceClient {
    public static void main(String[] args) {
cl1();cl2();
    }

    public static void cl1() {
        try {
            // 接口地址
            String address = "http://localhost:8899/services/CommonService?wsdl";
            // 代理工厂
            JaxWsProxyFactoryBean jaxWsProxyFactoryBean = new JaxWsProxyFactoryBean();
            // 设置代理地址
            jaxWsProxyFactoryBean.setAddress(address);
            // 设置接口类型
            jaxWsProxyFactoryBean.setServiceClass(CommonWebService.class);
            // 创建一个代理接口实现
            CommonWebService cs = (CommonWebService) jaxWsProxyFactoryBean.create();
            // 数据准备
            String userName = "xiaomin";
            // 调用代理接口的方法调用并返回结果
            String result = cs.sayHello(userName);
            System.out.println("返回结果:" + result);
        } catch (Exception e) {
            e.printStackTrace();
        }
    }

    /**
     * 动态调用方式
     */
    public static void cl2() {
        // 创建动态客户端
        JaxWsDynamicClientFactory dcf = JaxWsDynamicClientFactory.newInstance();
        Client client = dcf.createClient("http://localhost:8899/services/CommonService?wsdl");
        // 需要密码的情况需要加上用户名和密码
        // client.getOutInterceptors().add(new ClientLoginInterceptor(USER_NAME,
        // PASS_WORD));
        Object[] objects = new Object[0];
        try {
            // invoke("方法名",参数1,参数2,参数3....);
            objects = client.invoke("sayHello", "xiaohon");
            System.out.println("返回数据:" + objects[0]);
        } catch (java.lang.Exception e) {
            e.printStackTrace();
        }
    }
}
